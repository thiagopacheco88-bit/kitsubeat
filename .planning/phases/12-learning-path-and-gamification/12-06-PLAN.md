    **Create `src/lib/sfx.ts`**:
    ```ts
    let levelUpAudio: HTMLAudioElement | null = null;
    export function preloadLevelUpSFX(): void {
      if (typeof window === 'undefined') return;
      if (!levelUpAudio) {
        levelUpAudio = new Audio('/sounds/level-up.mp3');
        levelUpAudio.preload = 'auto';
      }
    }
    export function playLevelUpSFX(enabled: boolean): void {
      if (!enabled) return;
      preloadLevelUpSFX();
      if (!levelUpAudio) return;
      levelUpAudio.currentTime = 0;
      levelUpAudio.play().catch(() => { /* autoplay blocked — silent degrade */ });
    }
    export function triggerHaptic(enabled: boolean, pattern: number | number[] = [80, 40, 80]): void {
      if (!enabled) return;
      try {
        if (typeof navigator !== 'undefined' && 'vibrate' in navigator) {
          navigator.vibrate(pattern);
        }
      } catch { /* iOS Safari edge case — silent */ }
    }
    ```
