    **getJlptGapSummary(userId: string)** returning `JlptGapRow[]`:
    ```ts
    export interface JlptGapRow {
      jlpt_level: 'N5'|'N4'|'N3'|'N2'|'N1';
      total_count: number;   // all vocabulary_items rows with this jlpt_level
      mastered_count: number; // rows with user_vocab_mastery.state = 2 for this user
      known_count: number;    // state IN (1,2,3)
    }
    ```
    SQL (pin via drizzle's `sql` tagged template for the CASE ordering — do NOT over-engineer with two separate queries):
    ```sql
    SELECT
      vi.jlpt_level::text AS jlpt_level,
      COUNT(*)::int AS total_count,
      COUNT(*) FILTER (WHERE m.state = 2)::int AS mastered_count,
      COUNT(*) FILTER (WHERE m.state IN (1,2,3))::int AS known_count
    FROM vocabulary_items vi
    LEFT JOIN user_vocab_mastery m
      ON m.vocab_item_id = vi.id AND m.user_id = $1
    WHERE vi.jlpt_level IS NOT NULL
    GROUP BY vi.jlpt_level
    ORDER BY CASE vi.jlpt_level::text
      WHEN 'N5' THEN 1 WHEN 'N4' THEN 2 WHEN 'N3' THEN 3 WHEN 'N2' THEN 4 WHEN 'N1' THEN 5
    END ASC;
    ```
    Returns 5 rows max. If a tier has zero vocab_items in the DB (shouldn't happen but defensive), it's omitted — JlptGapSummary.tsx will render a "N/A — catalog incomplete" row for any missing tier.

    ```tsx
    import { getJlptGapSummary, type JlptGapRow } from '@/lib/db/queries';
    export async function JlptGapSummary({ userId }: { userId: string }) {
      const rows = await getJlptGapSummary(userId);
      const allTiers: JlptGapRow['jlpt_level'][] = ['N5','N4','N3','N2','N1'];
      return (
        <section className="rounded-xl border bg-card p-4 mb-6" aria-labelledby="jlpt-gap-heading">
          <h2 id="jlpt-gap-heading" className="text-lg font-semibold mb-3">JLPT Mastery</h2>
          <ul className="space-y-2">
            {allTiers.map(tier => {
              const row = rows.find(r => r.jlpt_level === tier);
              if (!row) return <li key={tier}>{tier}: catalog data not yet seeded</li>;
              const pct = row.total_count > 0 ? Math.round((row.mastered_count / row.total_count) * 100) : 0;
              const remaining = Math.max(0, row.total_count - row.mastered_count);
              return (
                <li key={tier} className="...">
                  <span>{tier}</span>
                  <span>{row.mastered_count} / {row.total_count} mastered</span>
                  <span>{remaining} to go to be {tier} fluent</span>
                  <progress value={pct} max={100} />
                </li>
              );
            })}
          </ul>
        </section>
      );
    }
    ```
