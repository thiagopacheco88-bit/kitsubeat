   ```ts
   const jlptLevels = Array.from(
     new Set(
       Object.values(vocabData)
         .map((v) => v.jlpt_level)
         .filter((l): l is string => typeof l === "string" && l.length > 0)
   ```

2. Collect the set of vocab_item_ids already in the queue so we can exclude them from the pool (preventing a distractor from matching a card that will appear later in the session):
   ```ts
   const queuedIds = new Set(items.map((i) => i.vocab_item_id));
   ```

3. For each level, fetch up to 50 rows of `VocabRow`-shaped data from `vocabulary_items` where `jlpt_level = <level>` AND `id NOT IN (queuedIds)`. Use a SINGLE combined query to minimize roundtrips:
   ```ts
   let jlptPools: Record<string, VocabRow[]> = {};
   // Guard prevents pulling the full vocabulary_items table for the
   // "empty queue, no levels" case (jlptLevels is only populated when
   // vocabData has items with a non-empty jlpt_level).
   if (jlptLevels.length > 0) {
     const rows = await db
       .select({
         id: vocabularyItems.id,
         dictionary_form: vocabularyItems.dictionary_form,
         reading: vocabularyItems.reading,
         romaji: vocabularyItems.romaji,
         part_of_speech: vocabularyItems.part_of_speech,
         jlpt_level: vocabularyItems.jlpt_level,
         meaning: vocabularyItems.meaning,
         mnemonic: vocabularyItems.mnemonic,
         kanji_breakdown: vocabularyItems.kanji_breakdown,
       })
       .from(vocabularyItems)
       .where(
         and(
           inArray(vocabularyItems.jlpt_level, jlptLevels),
           notInArray(vocabularyItems.id, Array.from(queuedIds))
         )
       );
     // Group by level, cap 50 per level
     for (const row of rows) {
       if (!row.jlpt_level) continue;
       const pool = jlptPools[row.jlpt_level] ?? [];
       if (pool.length < 50) {
         pool.push(row as VocabRow);
         jlptPools[row.jlpt_level] = pool;
       }
     }
   }
   ```
   (Import `and` and `notInArray` from `drizzle-orm` — `inArray` is already imported.)

   Rationale for doing this in one query instead of N: the pool size is small (≤ 50 × number-of-levels-in-queue), fully indexed on `jlpt_level`, and one roundtrip is cheaper than per-level fetches. We do not need a random sample — the first 50 per level is fine for distractor purposes; `pickDistractors` does its own dedup and correctness filtering.

**Step B — create `src/lib/review/distractors.ts`:**

Small helper that adapts `VocabRow` (from `/api/review/queue`) to the `VocabEntry` shape consumed by `pickDistractors`. Mirrors the mapping already in `src/app/songs/[slug]/components/ExerciseTab.tsx` lines 134-149 (the song-page JLPT-pool fetch path).

```ts
import type { VocabEntry } from "@/lib/types/lesson";
import type { VocabRow } from "@/app/api/review/queue/route";
// pickDistractors is imported from "@/lib/exercises/generator" elsewhere;
// VocabEntry itself is canonically exported from "@/lib/types/lesson".

/**
 * Adapts a VocabRow (from the /api/review/queue endpoint) to the VocabEntry
 * shape consumed by pickDistractors.
 *
 * Only distractor-relevant fields are populated; example_from_song and
 * additional_examples get safe defaults because pickDistractors does not read
 * them. Mirrors the pattern in ExerciseTab.tsx lines 134-149.
 */
export function vocabRowToVocabEntry(row: VocabRow): VocabEntry {
  return {
    surface: row.dictionary_form,
    reading: row.reading,
    romaji: row.romaji,
    part_of_speech: row.part_of_speech as VocabEntry["part_of_speech"],
    jlpt_level: (row.jlpt_level ?? "N5") as VocabEntry["jlpt_level"],
    meaning:
      typeof row.meaning === "string"
        ? { en: row.meaning }
        : (row.meaning as Record<string, string>),
    vocab_item_id: row.id,
    example_from_song: "",
    additional_examples: [],
  };
}
```

Keep the function pure and import-cheap — it's called per-card at render time.
