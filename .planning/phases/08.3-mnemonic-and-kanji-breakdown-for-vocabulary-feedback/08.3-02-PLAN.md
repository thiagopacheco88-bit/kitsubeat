Concretely:

```typescript
// scripts/types/enrich.ts
import { z } from "zod";
import { zodToJsonSchema } from "zod-to-json-schema";

// Reuse an EN/PT/ES Localizable shape — keep it inline here to avoid cross-file coupling.
const LocalizableSchema = z.object({
  en: z.string().min(1),
  "pt-BR": z.string().min(1),
  es: z.string().min(1),
});

const JlptLevelSchema = z.enum(["N5", "N4", "N3", "N2", "N1"]).nullable();

export const KanjiCharEntrySchema = z.object({
  char: z.string().refine((c) => [...c].length === 1, "must be exactly one kanji character"),
  meaning: LocalizableSchema,
  on_yomi: z.string(),           // kana; may be empty if truly none
  kun_yomi: z.string(),          // kana; empty string permitted
  jlpt_level: JlptLevelSchema,
  radical_hint: LocalizableSchema,
});

export const KanjiBreakdownSchema = z.object({
  characters: z.array(KanjiCharEntrySchema).min(1),
  compound_note: LocalizableSchema.optional(),
});

export const VocabEnrichmentSchema = z.object({
  mnemonic: LocalizableSchema,
  kanji_breakdown: KanjiBreakdownSchema.nullable(),
});

export type VocabEnrichment = z.infer<typeof VocabEnrichmentSchema>;

export const ENRICH_JSON_SCHEMA = zodToJsonSchema(VocabEnrichmentSchema, {
  name: "VocabEnrichment",
  $refStrategy: "none",
});
```

Key constraints enforced by the Zod schema:
- `char` is exactly one Unicode code point (use `[...c].length === 1` — NOT `c.length === 1` — to handle multi-byte characters).
- `kanji_breakdown` is nullable so kana-only words can pass validation with `null`.
- `compound_note` is `.optional()` so single-kanji words don't need to supply it.
- All `Localizable` fields require all three languages present and non-empty, matching the existing `meaning` column shape.

Do NOT import from `src/lib/types/lesson.ts` — `scripts/` uses a separate tsconfig and lives outside the Next build; duplicating the Zod shape here avoids a cross-boundary import. Keep the shapes structurally compatible.

Create `scripts/seed/11-enrich-vocab.ts` using the inline Anthropic Messages API pattern from `scripts/seed/03b-generate-content-sequential.ts` (NOT the Batch API used by `03`). Structure:

```typescript
// scripts/seed/11-enrich-vocab.ts
import { config } from "dotenv";
config({ path: ".env.local" });

import Anthropic from "@anthropic-ai/sdk";
import pLimit from "p-limit";
import { eq, isNull } from "drizzle-orm";
import { getDb } from "../../src/lib/db/index.js";
import { vocabularyItems } from "../../src/lib/db/schema.js";
import { VocabEnrichmentSchema, ENRICH_JSON_SCHEMA } from "../types/enrich.js";
import { localize, type Localizable } from "../../src/lib/types/lesson.js";

const MODEL = "claude-haiku-4-5";  // cost-efficient; upgrade to sonnet-4-6 if quality check fails
const CONCURRENCY = 5;
const MAX_TOKENS = 1024;
const KANJI_RE = /[\u4e00-\u9fff]/;

function buildPrompt(row: typeof vocabularyItems.$inferSelect, hasKanji: boolean): string {
  const meaningEn = localize(row.meaning as Localizable, "en");
  return `Generate learning aids for this Japanese vocabulary word.

Word: ${row.dictionary_form}
Reading: ${row.reading}
Meaning (EN): ${meaningEn}
Part of speech: ${row.part_of_speech}
${row.jlpt_level ? `JLPT level: ${row.jlpt_level}` : "JLPT level: unlisted"}

Return a JSON object with:

1. "mnemonic": { "en": "...", "pt-BR": "...", "es": "..." }
   - Each language: ONE short sentence, 10-15 words.
   - Tone: playful + visual. Use wordplay, vivid imagery, or sound-alikes to the Japanese reading.
   - NOT etymology. NOT anime references.

2. "kanji_breakdown": ${hasKanji ? `
   {
     "characters": [
       {
         "char": "<single kanji>",
         "meaning": { "en": "...", "pt-BR": "...", "es": "..." },
         "on_yomi": "<kana, empty string if none>",
         "kun_yomi": "<kana, empty string if none>",
         "jlpt_level": "N5" | "N4" | "N3" | "N2" | "N1" | null,
         "radical_hint": { "en": "<short string e.g. 'fire + person'>", "pt-BR": "...", "es": "..." }
       }
     ],
     "compound_note": { "en": "...", "pt-BR": "...", "es": "..." } // ONLY if 2+ kanji — bridge per-character meanings to the whole word
   }
   Output exactly one character entry per kanji in "${row.dictionary_form}".` : `null (this word has no kanji)`}

Output JSON only. No prose around it.`;
}

async function main() {
  const db = getDb();
  const client = new Anthropic({ apiKey: process.env.ANTHROPIC_API_KEY!, maxRetries: 3 });
  const limit = pLimit(CONCURRENCY);

  // Skip-already-done: mnemonic IS NULL is the only enrichment gate.
  const rows = await db.select().from(vocabularyItems).where(isNull(vocabularyItems.mnemonic));
  console.log(`[enrich-vocab] ${rows.length} rows pending enrichment`);

  let ok = 0, fail = 0;
  const tasks = rows.map((row) => limit(async () => {
    try {
      const hasKanji = KANJI_RE.test(row.dictionary_form);
      const response = await client.messages.create({
        model: MODEL,
        max_tokens: MAX_TOKENS,
        messages: [{ role: "user", content: buildPrompt(row, hasKanji) }],
        // Structured output — same pattern used by batch-claude.ts in this repo.
        // eslint-disable-next-line @typescript-eslint/no-explicit-any
        output_config: { format: { type: "json_schema", json_schema: ENRICH_JSON_SCHEMA as any } } as any,
      });
      const text = response.content.find((b) => b.type === "text")?.text ?? "";
      const json = JSON.parse(text);
      const parsed = VocabEnrichmentSchema.safeParse(json);
      if (!parsed.success) {
        console.warn(`[fail] ${row.dictionary_form} (${row.id}):`, parsed.error.issues);
        fail++;
        return;
      }

      // Per-row commit so resume works — if the script dies halfway, already-processed rows are persistent.
      await db.update(vocabularyItems)
        .set({
          mnemonic: parsed.data.mnemonic,
          kanji_breakdown: parsed.data.kanji_breakdown, // may be null for kana-only
        })
        .where(eq(vocabularyItems.id, row.id));

      ok++;
      if ((ok + fail) % 25 === 0) console.log(`[progress] ${ok} ok / ${fail} fail / ${rows.length} total`);
    } catch (err) {
      console.warn(`[error] ${row.dictionary_form} (${row.id}):`, err instanceof Error ? err.message : err);
      fail++;
    }
  }));

  await Promise.all(tasks);
  console.log(`[done] succeeded=${ok} failed=${fail}`);
  if (fail > 0) process.exit(1);
}

main().catch((e) => { console.error(e); process.exit(1); });
```

Key constraints:
- **Inline `messages.create`, NOT Batch API.** User preference is inline — do not add any `client.messages.batches` references.
- Per-row `db.update()` immediately after validation. Never batch updates. If the script dies on row 500/705, rerunning picks up at row 501 because those rows still have `mnemonic IS NULL`.
- Skip gate is strictly `isNull(vocabularyItems.mnemonic)` — not `isNull(kanji_breakdown)`, because kana-only words legitimately have `kanji_breakdown = null` forever.
- `p-limit` concurrency = 5 to avoid Anthropic 429s on ~705 rows.
- Kanji detection: `KANJI_RE.test(row.dictionary_form)` — reuses the same regex pattern already in `scripts/backfill-vocab-identity.ts` and `scripts/seed/06-qa-agent.ts`.
