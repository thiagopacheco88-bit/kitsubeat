`isRowMastered(row: KanaRow, scriptMastery: MasteryMap, script: Script): boolean`
- Empty-row guard: `isRowMastered({chars: []} as KanaRow, {}, "hiragana")` → true (defensive).
- a-row (5 chars), all 0 stars → false.
- a-row (5 chars), 4 chars at >= 5 stars (80% exactly) → true.   (uses Math.ceil(5 * 0.8) === 4)
- a-row (5 chars), 3 chars at >= 5 stars (60%) → false.
- ya-row (3 chars), 3 chars at 5 stars → true.
- ya-row (3 chars), 2 chars at 5 stars → false.   (Math.ceil(3 * 0.8) === 3)
- Reads via the script-correct key: for `script: "katakana"` it must look up `scriptMastery[c.katakana]`, NOT `c.hiragana`.

`computeUnlockedRows(rows: KanaRow[], scriptMastery: MasteryMap, script: Script): Set<string>`
- Empty-rows array → empty Set.
- Single row, all 0 stars → Set with that one row id (first row is always unlocked — this is the KANA-04 self-teaching mechanism in action).
- 27 rows (HIRAGANA_ROWS), no mastery → Set of size 1 ({"a"}).
- After fully mastering a-row, ka-row becomes unlocked → Set { "a", "ka" }.
- Strict sequential: if a is mastered, ka is mastered, but sa is NOT, then unlocked = {a, ka, sa} (sa unlocks because ka was mastered) and ta is NOT unlocked. The function MUST `break` when it hits the first non-mastered row.
- Idempotent: calling with the same input twice returns equal sets.

```typescript
import type { KanaRow, MasteryMap, Script } from "./types";
import { ROW_UNLOCK_MASTERY_PCT, ROW_UNLOCK_MIN_STARS } from "./chart";

/** KANA-03: clamp delta to [0, 10]. correct→+1, wrong→-2. */
export function applyStarDelta(current: number, correct: boolean): number {
  const next = correct ? current + 1 : current - 2;
  return Math.max(0, Math.min(10, next));
}

/** KANA-06 helper: row counts as "mastered" when >= 80% of chars are at >= 5 stars. */
export function isRowMastered(row: KanaRow, scriptMastery: MasteryMap, script: Script): boolean {
  if (row.chars.length === 0) return true;
  const threshold = Math.ceil(row.chars.length * ROW_UNLOCK_MASTERY_PCT);
  let masteredCount = 0;
  for (const c of row.chars) {
    const key = script === "hiragana" ? c.hiragana : c.katakana;
    if ((scriptMastery[key] ?? 0) >= ROW_UNLOCK_MIN_STARS) masteredCount += 1;
  }
  return masteredCount >= threshold;
}

/** KANA-06: derive unlocked-row set from mastery. Strictly monotonic, sequential. */
export function computeUnlockedRows(
  rows: KanaRow[], // expected sorted by row.order
  scriptMastery: MasteryMap,
  script: Script,
): Set<string> {
  const unlocked = new Set<string>();
  if (rows.length === 0) return unlocked;
  unlocked.add(rows[0].id);
  for (let i = 0; i < rows.length - 1; i++) {
    if (isRowMastered(rows[i], scriptMastery, script)) {
      unlocked.add(rows[i + 1].id);
    } else {
      break;
    }
  }
  return unlocked;
}
```

```typescript
import type { KanaChar, MasteryMap, Script } from "./types";

export interface EligibleChar {
  char: KanaChar;
  stars: number;
  weight: number;
  script: Script;          // include so the UI knows which glyph to render in mixed mode
}

/** KANA-05: stars→weight table. 0→10, 5→5, 10→1 (1/5 of weightFor(5)). */
export function weightFor(stars: number): number {
  const s = Math.max(0, Math.min(10, stars));
  if (s <= 0) return 10;
  if (s >= 10) return 1;
  return 10 - s;
}

/** O(n) prefix-sum sample. Caller MUST ensure pool.length >= 1. */
export function pickWeighted<T extends { weight: number }>(
  pool: T[],
  rng: () => number = Math.random,
): T {
  const total = pool.reduce((s, x) => s + x.weight, 0);
  let r = rng() * total;
  for (const x of pool) {
    r -= x.weight;
    if (r <= 0) return x;
  }
  return pool[pool.length - 1]; // floating-point safety
}

export function buildKanaSession(params: {
  script: Script | "mixed";
  mastery: { hiragana: MasteryMap; katakana: MasteryMap };
  unlockedRows: { hiragana: Set<string>; katakana: Set<string> };
  chart: KanaChar[];
  questionCount?: number;
  rng?: () => number;
}): EligibleChar[] {
  const { script, mastery, unlockedRows, chart, questionCount = 20, rng = Math.random } = params;
  const pool: EligibleChar[] = [];
  for (const c of chart) {
    if ((script === "hiragana" || script === "mixed") && unlockedRows.hiragana.has(c.rowId)) {
      const stars = mastery.hiragana[c.hiragana] ?? 0;
      pool.push({ char: c, stars, weight: weightFor(stars), script: "hiragana" });
    }
    if ((script === "katakana" || script === "mixed") && unlockedRows.katakana.has(c.rowId)) {
      const stars = mastery.katakana[c.katakana] ?? 0;
      pool.push({ char: c, stars, weight: weightFor(stars), script: "katakana" });
    }
  }
  if (pool.length === 0) return [];
  return Array.from({ length: questionCount }, () => pickWeighted(pool, rng));
}

export function buildDistractors(params: {
  correctRomaji: string;
  script: Script;
  unlockedRows: Set<string>;
  chart: KanaChar[];
  count?: number;
  rng?: () => number;
}): string[] {
  const { correctRomaji, script, unlockedRows, chart, count = 3, rng = Math.random } = params;
  void script; // reserved — distractors are romaji strings; script affects which rowIds are unlocked, already captured
  const romajiSet = new Set<string>();
  for (const c of chart) {
    if (unlockedRows.has(c.rowId) && c.romaji !== correctRomaji) romajiSet.add(c.romaji);
  }
  const pool = [...romajiSet];
  for (let i = pool.length - 1; i > 0; i--) {
    const j = Math.floor(rng() * (i + 1));
    [pool[i], pool[j]] = [pool[j], pool[i]];
  }
  return pool.slice(0, count);
}
```
