```typescript
/**
 * Japanese TTS helper — Web Speech API (browser-native).
 *
 * Phase 08.4: tap-to-play speaker icon on the LearnCard.
 * Zero dependencies, zero cost, ~near-zero latency.
 *
 * Graceful degradation: if no Japanese voice is available, callers should hide
 * the speaker icon entirely (check hasJapaneseVoice() before render). Do NOT
 * let the icon call speakJapanese() and silently do nothing — show, hide.
 *
 * Pitfall (research Area 9, #5): speechSynthesis.getVoices() may return [] on
 * first call in Chromium browsers. Voices load asynchronously and fire a
 * "voiceschanged" event. Components that rely on hasJapaneseVoice() must listen
 * for that event and re-render.
 */

/**
 * Returns true if the browser supports speech synthesis and has a Japanese voice.
 * Safe to call at module top-level or during render — returns false on SSR.
 */
export function hasJapaneseVoice(): boolean {
  if (typeof window === "undefined") return false;
  if (!("speechSynthesis" in window)) return false;
  const voices = window.speechSynthesis.getVoices();
  return voices.some((v) => v.lang.startsWith("ja"));
}

/**
 * Speaks the given Japanese text (kanji, kana, or mixed).
 * Always cancels any in-flight utterance first — rapid taps queue nothing.
 * No-op on SSR, on browsers without speechSynthesis, or if text is empty.
 *
 * Call hasJapaneseVoice() first to decide whether to show the speaker icon;
 * this function silently no-ops if the voice is missing rather than throwing.
 */
export function speakJapanese(text: string): void {
  if (typeof window === "undefined") return;
  if (!("speechSynthesis" in window)) return;
  if (!text) return;

  const voices = window.speechSynthesis.getVoices();
  const jaVoice = voices.find((v) => v.lang.startsWith("ja")) ?? null;

  const utt = new SpeechSynthesisUtterance(text);
  utt.lang = "ja-JP";
  if (jaVoice) utt.voice = jaVoice;

  // Cancel any current speech so rapid taps don't pile up.
  window.speechSynthesis.cancel();
  window.speechSynthesis.speak(utt);
}

/**
 * Registers a listener for the voiceschanged event. Returns an unsubscribe fn.
 * Use in a useEffect to re-check hasJapaneseVoice() after voices load lazily.
 * No-op + no-op unsubscribe on SSR.
 */
export function onVoicesChanged(cb: () => void): () => void {
  if (typeof window === "undefined") return () => {};
  if (!("speechSynthesis" in window)) return () => {};

  const handler = () => cb();
  window.speechSynthesis.addEventListener("voiceschanged", handler);
  return () => {
    window.speechSynthesis.removeEventListener("voiceschanged", handler);
  };
}
```

No unit tests for this file — it's a thin browser-API wrapper with no branching logic worth covering. Plan 04's E2E will verify the speaker icon renders when the voice exists.
